def reverseList(head):
    if not head or not head.next:
        return head
    new_head = reverseList(head.next)
    head.next.next = head
    head.next = None
    return new_head

Fast & Slow Pointers (Tortoise & Hare)
Linked list cycles · Find middle · Happy Number
▼
Signal: Linked list cycles, finding the middle node, finding the start of a cycle, "Happy Number"
def fast_slow(head):
    slow, fast = head, head

    while fast and fast.next:   # BOTH needed: fast alone may crash
        slow = slow.next
        fast = fast.next.next

        if slow == fast:
            return True  # Cycle detected

    return False

# ─── Find CYCLE START (Floyd's Phase 2) ───
def detect_cycle_start(head):
    slow, fast = head, head
    while fast and fast.next:
        slow = slow.next; fast = fast.next.next
        if slow == fast:
            slow2 = head
            while slow != slow2:       # both move 1 step
                slow = slow.next; slow2 = slow2.next
            return slow                # cycle entry point
    return None
Time: O(n) Space: O(1)
Complexity detail: Without cycle (find middle): fast visits ~n nodes → O(n). With cycle: proven worst-case ≤ 2n node visits → still O(n).

Two Pointers (Opposite Ends)
Sorted arrays · Two Sum · Palindrome check
▼
Signal: Sorted arrays, "Two Sum" in sorted array, reversing strings, checking palindromes, container with most water
def two_pointers(nums, target):
    l, r = 0, len(nums) - 1

    while l < r:
        curr = nums[l] + nums[r]
        if curr == target:
            return [l, r]
        elif curr < target:
            l += 1
        else:
            r -= 1
    return []

# ─── 3Sum (sort + two pointers) ───
def threeSum(nums):
    nums.sort(); res = []
    for i in range(len(nums) - 2):
        if i > 0 and nums[i] == nums[i-1]: continue  # skip dupes
        l, r = i + 1, len(nums) - 1
        while l < r:
            s = nums[i] + nums[l] + nums[r]
            if s < 0: l += 1
            elif s > 0: r -= 1
            else:
                res.append([nums[i], nums[l], nums[r]])
                while l < r and nums[l] == nums[l+1]: l += 1
                l += 1; r -= 1
    return res
Time: O(n) two-sum / O(n²) three-sumSpace: O(1)

Sliding Window (Variable Size)
Longest substring with condition · Max consecutive · Extreme limits
▼
Signal: "Longest/shortest subarray with condition", Max consecutive ones. The Goal: Finding Extreme Limits (longest, shortest, maximum)
def sliding_window(s):
    l = 0; ans = 0; count = {}

    for r in range(len(s)):
        # 1. Add right element
        count[s[r]] = count.get(s[r], 0) + 1

        # 2. Shrink left while invalid
        while not_valid(count):
            count[s[l]] -= 1
            if count[s[l]] == 0: del count[s[l]]
            l += 1

        # 3. Update answer
        ans = max(ans, r - l + 1)
    return ans

# ─── Fixed Size Window ───
def fixed_window(nums, k):
    window_sum = sum(nums[:k]); best = window_sum
    for i in range(k, len(nums)):
        window_sum += nums[i] - nums[i - k]
        best = max(best, window_sum)
    return best
Time: O(n)Space: O(k) — k = max distinct chars
Each element is visited at most twice: once by r (right), once by l (left) when shrinking → O(n) guaranteed.

Prefix Sum + Hash Map
Subarray sum = K · Count subarrays · Exact targets
▼
Signal: "Subarray sum equals K", "Number of subarrays with sum...", continuous subarray problems. The Goal: Finding Exact Targets
🚗 Road Trip Analogy
Imagine tracking your odometer while driving. At 1:00 PM = 50 miles. At 3:00 PM = 150 miles. Since 150 − 50 = 100, you drove exactly 100 miles in between. That's the key: curr_sum − prev_sum = k means curr_sum − k must exist in history. 
def subarray_sum(nums, k):
    prefix_map = {0: 1}  # handles subarray starting at 0
    curr_sum = count = 0

    for num in nums:
        curr_sum += num
        if (curr_sum - k) in prefix_map:
            count += prefix_map[curr_sum - k]
        prefix_map[curr_sum] = prefix_map.get(curr_sum, 0) + 1

    return count
Time: O(n)Space: O(n)
📝 Sliding Window vs Prefix Sum: Sliding Window finds the optimal range (Max/Min). Prefix Sum uses a historical map to count every exact "gap" equal to target k.

Binary Search on Answer Space
Minimize the maximum · Koko eating bananas · Capacity to ship
▼
Signal: "Minimize the maximum", "Koko eating bananas", "Capacity to ship packages". You are NOT searching the array — you are searching the range of possible answers.
def solve():
    def feasible(val):
        return ...  # True if val is sufficient/possible

    l, r = 1, 10**9
    while l < r:
        mid = l + (r - l) // 2
        if feasible(mid):
            r = mid        # Try smaller
        else:
            l = mid + 1   # Need bigger
    return l
THE 2 RULES — memorize forever:
• Finding a specific number? → while l <= r
• Finding minimum possible value? → while l < r → return l

THE ONE SENTENCE: If mid could be the answer → r = mid. If mid is definitely NOT the answer → mid ± 1.
Integer Overflow (Java/C++): Use mid = low + (high - low) / 2 not (low + high) / 2. Python handles big ints natively but use the safe form to show you understand.
Time: O(log N × cost of feasible)Space: O(1)

Modified Binary Search (Rotated / Tricky)
Rotated sorted array · Mountain array · First/Last position
▼
Signal: Sorted array that was rotated, searching in a mountain array, first/last position of element
def search_rotated(nums, target):
    l, r = 0, len(nums) - 1

    while l <= r:
        mid = (l + r) // 2
        if nums[mid] == target: return mid

        # Determine which side is sorted
        if nums[l] <= nums[mid]:   # Left side is sorted
            if nums[l] <= target < nums[mid]: r = mid - 1
            else: l = mid + 1
        else:                       # Right side is sorted
            if nums[mid] < target <= nums[r]: l = mid + 1
            else: r = mid - 1
    return -1

# ─── Find Min in Rotated Array ───
def findMin(nums):
    l, r = 0, len(nums) - 1
    while l < r:
        mid = (l + r) // 2
        if nums[mid] > nums[r]: l = mid + 1  # min is in right half
        else: r = mid                        # min is in left half (or mid)
    return nums[l]
Time: O(log N)Space: O(1)

Top 'K' Elements (Heaps)

import heapq

def find_k_largest(nums, k):
    heap = []  # Min-heap (Python default)
    for n in nums:
        heapq.heappush(heap, n)
        if len(heap) > k:
            heapq.heappop(heap)  # Remove smallest → keeps K largest
    return heap[0]  # Kth largest

# ─── Merge K Sorted Lists ───
def mergeKLists(lists):
    heap = []; dummy = curr = ListNode(0)
    for i, l in enumerate(lists):
        if l: heapq.heappush(heap, (l.val, i, l))
    while heap:
        val, i, node = heapq.heappop(heap)
        curr.next = node; curr = curr.next
        if node.next: heapq.heappush(heap, (node.next.val, i, node.next))
    return dummy.next

# ─── Median from Data Stream (Two Heaps) ───
class MedianFinder:
    def __init__(self):
        self.small = []  # Max-heap: smaller half (negate values)
        self.large = []  # Min-heap: larger half

    def addNum(self, num):
        heapq.heappush(self.small, -num)
        heapq.heappush(self.large, -heapq.heappop(self.small))
        if len(self.large) > len(self.small):
            heapq.heappush(self.small, -heapq.heappop(self.large))

    def findMedian(self):
        if len(self.small) > len(self.large):
            return -self.small[0]
        return (-self.small[0] + self.large[0]) / 2.0
⚠️ Heap warning: Inside a min-heap, ONLY heap[0] is guaranteed to be the smallest. Everything else is unordered — never return heap[k].

def isValidBST(root, left=float('-inf'), right=float('inf')):
    if not root: return True
    if not (left < root.val < right): return False
    return (isValidBST(root.left, left, root.val) and
            isValidBST(root.right, root.val, right))

# ─── Inorder Approach (values must be strictly increasing) ───
def isValidBST_inorder(root):
    prev = [float('-inf')]
    def inorder(node):
        if not node: return True
        if not inorder(node.left): return False
        if node.val <= prev[0]: return False
        prev[0] = node.val
        return inorder(node.right)
    return inorder(root)

BFS (Level Order Traversal)
Shortest path unweighted · Level order · Rotting oranges
▼
Signal: "Shortest path in unweighted graph", "Level order", "Nearest gate/rotting orange"
from collections import deque

def bfs(root):
    if not root: return []
    q = deque([root]); result = []

    while q:
        level = []
        for _ in range(len(q)):  # snapshot level size BEFORE adding children
            node = q.popleft()
            level.append(node.val)
            if node.left: q.append(node.left)
            if node.right: q.append(node.right)
        result.append(level)
    return result

# ─── Graph BFS (shortest path) ───
def shortest_path(graph, start, end):
    q = deque([(start, 0)]); visited = {start}
    while q:
        node, dist = q.popleft()
        if node == end: return dist
        for nei in graph[node]:
            if nei not in visited:
                visited.add(nei); q.append((nei, dist + 1))
    return -1
Why range(len(q))? Without it, you can't tell where one level ends and the next begins — children are added while you're still processing the current level. The for loop's end = the level boundary.
Time: O(N)Space: O(W) — W = max width

DFS — Backtracking
Generate subsets · Combinations · Permutations · Grid search
▼

📦 Combinations / Subsets

Signal: "Generate all subsets", "Combination Sum". Input [1,2,3] → Output [[], [1], [2], ...]
def subsets(nums):
    res = []
    def backtrack(start, path):
        res.append(path[:])  # Must copy! not reference
        for i in range(start, len(nums)):
            path.append(nums[i])         # 1. Choose
            backtrack(i + 1, path)       # 2. Explore (i+1: no going backward)
            path.pop()                   # 3. Un-choose (backtrack)
    backtrack(0, [])
    return res

# ─── With duplicates: sort + skip ───
def subsetsWithDup(nums):
    nums.sort(); res = []
    def backtrack(start, path):
        res.append(path[:])
        for i in range(start, len(nums)):
            if i > start and nums[i] == nums[i-1]: continue  # skip dupes
            path.append(nums[i]); backtrack(i+1, path); path.pop()
    backtrack(0, []); return res
Time: O(n × 2ⁿ)Space: O(n)
Why 2ⁿ subsets? Each element has 2 choices: include or exclude. 2 × 2 × ... (n times) = 2ⁿ. The n multiplier comes from the O(n) cost of path[:] copy.

🔀 Permutations

Signal: "All possible arrangements", "Rankings". Input [1,2,3] → [[1,2,3],[1,3,2],...]
def permutations(nums):
    result = []; used = [False] * len(nums)
    def backtrack(path):
        if len(path) == len(nums):
            result.append(path[:]); return
        for i in range(len(nums)):
            if used[i]: continue  # skip already chosen
            used[i] = True; path.append(nums[i])
            backtrack(path)
            path.pop(); used[i] = False
    backtrack([]); return result
Time: O(n × n!)Space: O(n)

🗺️ Grid / Word Search

Signal: "Word Search", "Number of Islands", "Maze"
def grid_search(grid, word):
    rows, cols = len(grid), len(grid[0])
    def backtrack(r, c, index):
        if index == len(word): return True
        if (r < 0 or r >= rows or c < 0 or c >= cols or
            grid[r][c] != word[index]): return False

        temp = grid[r][c]; grid[r][c] = "#"  # mark visited
        found = (backtrack(r+1,c,index+1) or backtrack(r-1,c,index+1) or
                 backtrack(r,c+1,index+1) or backtrack(r,c-1,index+1))
        grid[r][c] = temp  # restore
        return found

    for r in range(rows):
        for c in range(cols):
            if backtrack(r, c, 0): return True
    return False
Time: O(N × M × 3ᴸ)Space: O(L) — L = word length
📝 Backtracking Decision Tree:
Subsets: start = i+1 (no reuse, no reorder)
Combination Sum (unlimited): start = i (reuse ok)
Permutations: used[] array (reorder ok, no reuse)

DFS on Trees (Bottom-Up State)
Diameter · Is Balanced · Max Path Sum
▼
Signal: "Diameter of tree", "Is Balanced", "Max Path Sum". Key: ask child for info, process it, return info to parent.
def max_path_sum(root):
    global_max = float('-inf')

    def dfs(node):
        nonlocal global_max
        if not node: return 0

        left  = max(dfs(node.left),  0)  # ignore negative paths
        right = max(dfs(node.right), 0)

        global_max = max(global_max, node.val + left + right)  # "split" point
        return node.val + max(left, right)  # return one side only

    dfs(root)
    return global_max

# ─── Diameter of Binary Tree ───
def diameterOfBinaryTree(root):
    diameter = 0
    def dfs(node):
        nonlocal diameter
        if not node: return 0
        left = dfs(node.left); right = dfs(node.right)
        diameter = max(diameter, left + right)  # edges, not nodes
        return 1 + max(left, right)
    dfs(root); return diameter
Python scoping rule: For int/str (immutable) → use nonlocal to modify. For list/dict (mutable) → you can mutate directly without nonlocal.
Time: O(n)Space: O(h) — h = tree height

Topological Sort (Kahn's Algorithm)
Course schedule · Project dependencies · Build order
▼
Signal: "Course Schedule", "Project Dependencies", "Build order", "Alien Dictionary"
from collections import defaultdict, deque

def topo_sort(n, edges):
    graph = defaultdict(list)
    indegree = {i: 0 for i in range(n)}

    for u, v in edges:
        graph[u].append(v)
        indegree[v] += 1  # u→v means only v's indegree increases

    q = deque([node for node in indegree if indegree[node] == 0])
    result = []

    while q:
        node = q.popleft(); result.append(node)
        for nei in graph[node]:
            indegree[nei] -= 1
            if indegree[nei] == 0: q.append(nei)

    return result if len(result) == n else []  # empty = cycle detected
Cycle detection: If graph has a cycle → some nodes never reach indegree 0 → len(result) < n → return [].
Time: O(V + E)Space: O(V + E)

Union-Find (Disjoint Set Union)
Connected components · Redundant connection · Graph valid tree
▼
Signal: "Connected components", "Number of Islands 2", "Redundant Connection", "Graph Valid Tree"
class UnionFind:
    def __init__(self, n):
        self.parent = list(range(n))  # index = child, value = parent
        self.rank = [1] * n
        self.components = n

    def find(self, n1):  # find root
        while n1 != self.parent[n1]:
            self.parent[n1] = self.parent[self.parent[n1]]  # path compression
            n1 = self.parent[n1]
        return n1

    def union(self, n1, n2):
        p1, p2 = self.find(n1), self.find(n2)
        if p1 == p2: return False  # already connected
        if self.rank[p1] > self.rank[p2]:   self.parent[p2] = p1
        elif self.rank[p1] < self.rank[p2]: self.parent[p1] = p2
        else: self.parent[p2] = p1; self.rank[p1] += 1
        self.components -= 1
        return True

    def connected(self, n1, n2):
        return self.find(n1) == self.find(n2)
Two key optimizations:
1. Path Compression: Each node skips to its grandparent → flattens tree over time → future lookups O(1).
2. Union by Rank: Attach shorter tree to taller → keeps height O(log N).
Time: O(α(N)) ≈ O(1)Space: O(N)

Monotonic Stack
Next greater/smaller · Largest rectangle · Daily temperatures
▼
Signal: "Next Greater Element", "Next Smaller Element", "Largest Rectangle in Histogram", "Daily Temperatures"
def next_greater_element(nums):
    stack = []           # stores indices (decreasing values)
    res = [-1] * len(nums)

    for i, num in enumerate(nums):
        while stack and nums[stack[-1]] < num:
            index = stack.pop()
            res[index] = num
        stack.append(i)
    return res

# ─── Largest Rectangle in Histogram ───
def largestRectangleArea(heights):
    stack = []; max_area = 0
    for i, h in enumerate(heights + [0]):  # append 0 to flush stack
        while stack and heights[stack[-1]] >= h: # A smaller h acts as a right boundary
            height = heights[stack.pop()]
            width = i if not stack else i - stack[-1] - 1
            max_area = max(max_area, height * width)
        stack.append(i)
    return max_area
Optimization: Using `>` is correct, but `>=` is an optimization since the latest duplicate height covers the full width, making earlier identical heights redundant.
Time: O(n) — each element pushed/popped at most onceSpace: O(n)

Trie (Prefix Tree)
Autocomplete · Word Search II · Prefix matching
▼
Signal: "Autocomplete", "Word Search II", "Prefix matching"
class TrieNode:
    def __init__(self):
        self.children = {}
        self.is_end = False

class Trie:
    def __init__(self): self.root = TrieNode()

    def insert(self, word):
        curr = self.root
        for char in word:
            if char not in curr.children:
                curr.children[char] = TrieNode()
            curr = curr.children[char]
        curr.is_end = True

    def search(self, word):
        node = self._find(word)
        return node is not None and node.is_end

    def startsWith(self, prefix):
        return self._find(prefix) is not None

    def _find(self, prefix):
        curr = self.root
        for char in prefix:
            if char not in curr.children: return None
            curr = curr.children[char]
        return curr

DP: 0/1 Knapsack
Partition equal subset · Target sum · Coin change
▼
Signal: "Partition Equal Subset Sum", "Target Sum", "Coin Change 2"
def knapsack(nums, target):
    dp = [False] * (target + 1)
    dp[0] = True  # base case: sum of 0 is always achievable

    for num in nums:
        for i in range(target, num - 1, -1):  # BACKWARDS: prevents reuse
            dp[i] = dp[i] or dp[i - num]
    return dp[target]

# ─── Coin Change (minimum coins = unbounded knapsack) ───
def coinChange(coins, amount):
    dp = [float('inf')] * (amount + 1)
    dp[0] = 0
    for coin in coins:
        for i in range(coin, amount + 1):  # FORWARD: allows reuse
            dp[i] = min(dp[i], dp[i - coin] + 1)
    return dp[amount] if dp[amount] != float('inf') else -1
⚠️ Why reverse iteration?
Forward → Unbounded Knapsack (item used multiple times): dp[i - num] reads an already-updated value from this loop → chain reaction.
Backward → 0/1 Knapsack (item used once): dp[i - num] reads from the previous item's iteration → no reuse.
Time: O(n × target)Space: O(target)

🎯 Greedy Insight
When we pop the minimum-cost node from the heap, we know it's final — no future path can be cheaper. That's why Dijkstra is a greedy algorithm.
import heapq
from collections import defaultdict

def dijkstra(n, edges, start_node):
    graph = defaultdict(list)
    for u, v, w in edges:
        graph[u].append((v, w))

    min_heap = [(0, start_node)]
    shortest = {}

    while min_heap:
        w1, n1 = heapq.heappop(min_heap)
        if n1 in shortest: continue  # lazy deletion: skip stale entries
        shortest[n1] = w1

        for n2, w2 in graph[n1]:
            if n2 not in shortest:
                heapq.heappush(min_heap, (w1 + w2, n2))

    return shortest

class Node:
    def __init__(self, key, val):
        self.key, self.val = key, val
        self.prev = self.next = None

class LRUCache:
    def __init__(self, capacity):
        self.cap = capacity; self.cache = {}
        self.left, self.right = Node(0,0), Node(0,0)  # dummy sentinels
        self.left.next, self.right.prev = self.right, self.left

    def remove(self, node):
        prev, nxt = node.prev, node.next
        prev.next, nxt.prev = nxt, prev

    def insert(self, node):  # insert at right (Most Recent)
        prev, nxt = self.right.prev, self.right
        prev.next = nxt.prev = node
        node.next, node.prev = nxt, prev

    def get(self, key):
        if key in self.cache:
            self.remove(self.cache[key]); self.insert(self.cache[key])
            return self.cache[key].val
        return -1

    def put(self, key, value):
        if key in self.cache: self.remove(self.cache[key])
        self.cache[key] = Node(key, value); self.insert(self.cache[key])
        if len(self.cache) > self.cap:
            lru = self.left.next  # evict LRU (leftmost real node)
            self.remove(lru); del self.cache[lru.key]

def max_sliding_window(nums, k):
    output = []; q = deque()  # stores INDICES

    for r in range(len(nums)):
        while q and nums[q[-1]] < nums[r]:  # pop smaller (useless)
            q.pop()
        q.append(r)

def solve_bitmask(n, costs):
    memo = {}
    def dfs(i, mask):
        if i == n: return 0
        if (i, mask) in memo: return memo[(i, mask)]

        res = float('inf')
        for job in range(n):
            if not (mask & (1 << job)):  # job not yet assigned
                res = min(res, costs[i][job] + dfs(i+1, mask | (1 << job)))
        memo[(i, mask)] = res; return res

    return dfs(0, 0)

Segment Tree
Range sum/max queries + point updates
▼
Signal: "Find sum/max of range [L, R]" AND "Update value at index i". Why not prefix sum? Prefix sum is O(N) to update. Segment Tree is O(log N) for both.
class SegmentTree:
    def __init__(self, data):
        self.n = len(data)
        self.tree = [0] * (2 * self.n)
        for i in range(self.n): self.tree[self.n + i] = data[i]
        for i in range(self.n - 1, 0, -1):
            self.tree[i] = self.tree[2*i] + self.tree[2*i+1]

    def update(self, i, val):
        i += self.n; self.tree[i] = val
        while i > 1:
            i //= 2
            self.tree[i] = self.tree[2*i] + self.tree[2*i+1]

    def query(self, l, r):  # Range [l, r)
        l += self.n; r += self.n; res = 0
        while l < r:
            if l % 2 == 1: res += self.tree[l]; l += 1
            if r % 2 == 1: r -= 1; res += self.tree[r]
            l //= 2; r //= 2
        return res
Query/Update: O(log N)Space: O(N)

def rolling_hash(s, length):
    base = 26; mod = 2**63 - 1
    curr_hash = 0
    for i in range(length):
        curr_hash = (curr_hash * base + ord(s[i])) % mod

    seen = {curr_hash}

    dist[src] = 0

    return dist

                dp[i] = max(dp[i], dp[j] + 1)
    return max(dp)

String DP (Edit Distance / LCS)
Edit distance · Longest common subsequence · Regex matching
▼
Signal: "Edit Distance", "Longest Common Subsequence", "Regex Matching". Two strings → 2D DP where dp[i][j] = answer for s1[:i] and s2[:j].
def longestCommonSubsequence(text1, text2):
    m, n = len(text1), len(text2)
    dp = [[0] * (n + 1) for _ in range(m + 1)]

    for i in range(1, m + 1):
        for j in range(1, n + 1):
            if text1[i-1] == text2[j-1]:
                dp[i][j] = dp[i-1][j-1] + 1  # match: take diagonal + 1
            else:
                dp[i][j] = max(dp[i-1][j], dp[i][j-1])  # skip one char
    return dp[m][n]

# ─── Edit Distance ───
def minDistance(word1, word2):
    m, n = len(word1), len(word2)
    dp = [[0] * (n+1) for _ in range(m+1)]
    for i in range(m+1): dp[i][0] = i
    for j in range(n+1): dp[0][j] = j

    for i in range(1, m+1):
        for j in range(1, n+1):
            if word1[i-1] == word2[j-1]:
                dp[i][j] = dp[i-1][j-1]
            else:
                dp[i][j] = 1 + min(dp[i-1][j], dp[i][j-1], dp[i-1][j-1])
    return dp[m][n]
Time: O(m × n)Space: O(m × n)

# ─── Jump Game ───
def canJump(nums):
    max_reach = 0
    for i, n in enumerate(nums):
        if i > max_reach: return False
        max_reach = max(max_reach, i + n)
    return True

# ─── Jump Game II (min jumps) ───
def jump(nums):
    jumps = cur_end = cur_far = 0
    for i in range(len(nums) - 1):
        cur_far = max(cur_far, i + nums[i])
        if i == cur_end:
            jumps += 1; cur_end = cur_far
    return jumps

# ─── Non-overlapping Intervals (activity selection) ───
def eraseOverlapIntervals(intervals):
    intervals.sort(key=lambda x: x[1])  # sort by END time
    count = 0; prev_end = float('-inf')
    for start, end in intervals:
        if start >= prev_end:
            prev_end = end
        else:
            count += 1  # remove this interval
    return count

Graph Coloring / Bipartite Check
Is graph bipartite · Possible bipartition
▼
Signal: "Is graph bipartite?", "Can we split into two groups?". BFS/DFS coloring — alternate colors, if conflict → not bipartite.
def isBipartite(graph):
    n = len(graph); color = [-1] * n

    for start in range(n):
        if color[start] != -1: continue
        q = deque([start]); color[start] = 0
        while q:
            node = q.popleft()
            for nei in graph[node]:
                if color[nei] == -1:
                    color[nei] = 1 - color[node]; q.append(nei)
                elif color[nei] == color[node]:
                    return False
    return True

Lazy Propagation (Segment Tree)
Range updates + range queries
▼
Signal: "Range updates + range queries", "Too many updates to apply individually". Delay applying updates until absolutely necessary.
class LazySegmentTree:
    def __init__(self, arr):
        self.n = len(arr)
        self.tree = [0] * (4 * self.n)
        self.lazy = [0] * (4 * self.n)
        self._build(arr, 0, 0, self.n - 1)

    def _build(self, arr, node, start, end):
        if start == end:
            self.tree[node] = arr[start]; return
        mid = (start + end) // 2
        self._build(arr, 2*node+1, start, mid)
        self._build(arr, 2*node+2, mid+1, end)
        self.tree[node] = self.tree[2*node+1] + self.tree[2*node+2]

    def _push_down(self, node, start, end):
        if self.lazy[node] != 0:
            mid = (start + end) // 2
            self.tree[2*node+1] += self.lazy[node] * (mid - start + 1)
            self.tree[2*node+2] += self.lazy[node] * (end - mid)
            self.lazy[2*node+1] += self.lazy[node]
            self.lazy[2*node+2] += self.lazy[node]
            self.lazy[node] = 0
Time: O(log N) per operationSpace: O(N)

from sortedcontainers import SortedList

def containsNearbyAlmostDuplicate(nums, indexDiff, valueDiff):
    sl = SortedList()
    for i, num in enumerate(nums):
        if i > indexDiff: sl.remove(nums[i - indexDiff - 1])
        pos = sl.bisect_left(num - valueDiff)
        if pos < len(sl) and sl[pos] <= num + valueDiff:
            return True
        sl.add(num)
    return False

A* Search
Heuristic shortest path · Sliding puzzle
▼
Signal: "Sliding puzzle", "Shortest path with known target location". Dijkstra + heuristic. Priority = g(cost so far) + h(estimated cost to goal).
def a_star(grid, start, goal):
    def heuristic(a, b):  # Manhattan distance
        return abs(a[0]-b[0]) + abs(a[1]-b[1])

    heap = [(heuristic(start, goal), 0, start)]  # (f, g, node)
    visited = set()

    while heap:
        f, g, node = heapq.heappop(heap)
        if node == goal: return g
        if node in visited: continue
        visited.add(node)

        for dr, dc in [(0,1),(0,-1),(1,0),(-1,0)]:
            nr, nc = node[0]+dr, node[1]+dc
            if valid(nr, nc) and (nr, nc) not in visited:
                new_g = g + 1
                heapq.heappush(heap, (new_g + heuristic((nr,nc), goal), new_g, (nr,nc)))
    return -1
Time: O(E log V) with admissible heuristicSpace: O(V)

# LC 309: Best Time to Buy/Sell with Cooldown
def maxProfit(prices):
    if not prices: return 0
    n = len(prices)
    hold = -prices[0]  # max profit while holding stock
    sold = 0            # max profit just sold
    rest = 0            # max profit resting (cooldown)

    for i in range(1, n):
        prev_hold, prev_sold, prev_rest = hold, sold, rest
        hold = max(prev_hold, prev_rest - prices[i])  # keep or buy
        sold = prev_hold + prices[i]                   # sell
        rest = max(prev_rest, prev_sold)               # rest or cooldown

    return max(sold, rest)

# LC 123: At most K transactions
# dp[k][0] = max profit with k transactions, not holding
# dp[k][1] = max profit with k transactions, holding
def maxProfit_k(k, prices):
    if not prices: return 0
    dp = [[0, float('-inf')] for _ in range(k + 1)]
    for price in prices:
        for j in range(k, 0, -1):
            dp[j][0] = max(dp[j][0], dp[j][1] + price)
            dp[j][1] = max(dp[j][1], dp[j-1][0] - price)
    return dp[k][0]
Time: O(n × k)Space: O(k)
